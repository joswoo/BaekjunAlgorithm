import java.util.Scanner;

public class b04_1110 {
	static void Cycle() {
		System.out.print("숫자 : ");
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int num = n;
		int count = 0;
		
		do {
			num = (num%10)*10 + (num/10 + num%10)%10;
			System.out.println(num);
			count++;
		}while(num != n);
		System.out.println(count);
	}
	public static void main(String[] args) {
		Cycle();
	}
}
